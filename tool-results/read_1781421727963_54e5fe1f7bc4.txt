   334→  1890→/**
   335→  1891→ * Look up a translation key in the given locale.
   336→  1892→ * Falls back to English if the key is missing in the target locale.
   337→  1893→ * Returns the key itself if no translation is found at all.
   338→  1894→ */
   339→  1895→export function t(
   340→  1896→  key: string,
   341→  1897→  locale: Locale,
   342→  1898→  params?: Record<string, string | number>,
   343→  1899→): string {
   344→  1900→  const entry = translations[key];
   345→  1901→  if (!entry) return key;
   346→  1902→
   347→  1903→  let text = entry[locale] ?? entry['en'] ?? key;
   348→  1904→
   349→  1905→  if (params) {
   350→  1906→    for (const [k, v] of Object.entries(params)) {
   351→  1907→      text = text.replace(new RegExp(`\\{${k}\\}`, 'g'), String(v));
   352→  1908→    }
   353→  1909→  }
   354→  1910→
   355→  1911→  return text;
