  1514→// ─── Translation Function ─────────────────────────────────────────────────────
  1515→
  1516→/**
  1517→ * Look up a translation key in the given locale.
  1518→ * Falls back to English if the key is missing in the target locale.
  1519→ * Returns the key itself if no translation is found at all.
  1520→ */
  1521→export function t(
  1522→  key: string,
  1523→  locale: Locale,
  1524→  params?: Record<string, string | number>,
  1525→): string {
  1526→  const entry = translations[key];
  1527→  if (!entry) return key;
  1528→
  1529→  let text = entry[locale] ?? entry['en'] ?? key;
  1530→
  1531→  if (params) {
  1532→    for (const [k, v] of Object.entries(params)) {
  1533→      text = text.replace(new RegExp(`\\{${k}\\}`, 'g'), String(v));
  1534→    }
  1535→  }
  1536→
  1537→  return text;
