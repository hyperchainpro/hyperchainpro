  2229→/**
  2230→ * Look up a translation key in the given locale.
  2231→ * Falls back to English if the key is missing in the target locale.
  2232→ * Returns the key itself if no translation is found at all.
  2233→ */
  2234→export function t(
  2235→  key: string,
  2236→  locale?: Locale | null,
  2237→  params?: Record<string, string | number>,
  2238→): string {
  2239→  // Defensive: if locale is undefined/null, default to 'en'
  2240→  const loc: Locale = (locale as Locale) ?? 'en';
  2241→  const entry = translations[key];
  2242→  if (!entry) return key;
  2243→
  2244→  let text = entry[loc] ?? entry['en'] ?? key;
  2245→
  2246→  if (params) {
  2247→    for (const [k, v] of Object.entries(params)) {
  2248→      text = text.replace(new RegExp(`\\{${k}\\}`, 'g'), String(v));
  2249→    }
  2250→  }
  2251→
  2252→  return text;
